132void* async(void* arguments){
133 //printf("Start sync thread\n");
134 float transform_state = 0;
135 uint16_t total_time_us = 0;
136
137 struct async_args *args = arguments;
138 //printf("args %f - %i - %i\n", args->step_inc, args->step_delay_us, args->channel);
139
140 pthread_setcancelstate(PTHREAD_CANCEL_ENABLE, NULL);
141 pthread_setcanceltype(PTHREAD_CANCEL_ASYNCHRONOUS, NULL);
142 pthread_detach(pthread_self());
143
144 while( total_time_us < args->length_us){
145 switch ( args->type ) {
146 case PITCH_MOD_LINEAR:
147 chFreq[ args->channel ] =
148 notePhase[args->note_start]*(1-transform_state) +
149 notePhase[args->note_end]*transform_state;
150 if (transform_state >= 1) total_time_us = args->length_us; // break out of main while loop
151 break;
152 case PITCH_MOD_SINE:
153 float sine_val = waves_sine12_get((uint16_t)(transform_state*0xfff));
154 //printf("sine: %f\n", sine_val);
155 chFreq[ args->channel ] =
156 notePhase[args->note_start]*(1-sine_val) +
157 notePhase[args->note_end]*sine_val;
158 break;
159 case PITCH_MOD_RANDOM:
160 if( transform_state>1 ) transform_state--;
161 float rndTemp = (float)((rand()&0xfff)*transform_state)/0xfff;
162 chFreq[ args->channel ] =
163 notePhase[args->note_start]*(1-rndTemp) +
164 notePhase[args->note_end]*rndTemp;
165 break;
166
167 }
168
169 SDL_Delay(args->step_delay_us);
170 total_time_us += args->step_delay_us;
171 transform_state += args->step_inc;
172 }
173 chFreq[ args->channel ] = 0;
174 //printf("EXIT sync thread\n");
175 pthread_exit(NULL);
176}
