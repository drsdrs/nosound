134void* async( void* arguments ) {
135 // printf("Start sync thread\n");
136 float transform_state = 0;
137 uint16_t total_time_us = 0;
138
139 struct async_args* args = arguments;
140 // printf("args %f - %i - %i\n", args->step_inc, args->step_delay_us, args->channel);
141
142 pthread_setcancelstate( PTHREAD_CANCEL_ENABLE, NULL );
143 pthread_setcanceltype( PTHREAD_CANCEL_ASYNCHRONOUS, NULL );
144 pthread_detach( pthread_self() );
145
146 while ( total_time_us < args->length_us ) {
147 switch ( args->type ) {
148 case PITCH_MOD_LINEAR:
149 chFreq [ args->channel ] =
150 notePhase [ args->note_start ] * ( 1 - transform_state ) +
151 notePhase [ args->note_end ] * transform_state;
152 if ( transform_state >= 1 )
153 total_time_us = args->length_us; // break out of main while loop
154 break;
155 case PITCH_MOD_SINE:
156 float sine_val = waves_sine12_get( (uint16_t)( transform_state * 0xfff ) );
157 // printf("sine: %f\n", sine_val);
158 chFreq [ args->channel ] = notePhase [ args->note_start ] * ( 1 - sine_val ) +
159 notePhase [ args->note_end ] * sine_val;
160 break;
161 case PITCH_MOD_RANDOM:
162 if ( transform_state > 1 ) transform_state--;
163 float rndTemp = (float)( ( rand() & 0xfff ) * transform_state ) / 0xfff;
164 chFreq [ args->channel ] = notePhase [ args->note_start ] * ( 1 - rndTemp ) +
165 notePhase [ args->note_end ] * rndTemp;
166 break;
167 }
168
169 SDL_Delay( args->step_delay_us );
170 total_time_us += args->step_delay_us;
171 transform_state += args->step_inc;
172 }
173 chFreq [ args->channel ] = 0;
174 // printf("EXIT sync thread\n");
175 pthread_exit( NULL );
176}
