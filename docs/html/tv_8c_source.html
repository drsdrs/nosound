194 SDL_Event e;
195 if (SDL_PollEvent(&e) == 0) return;

197 if (e.type == SDL_QUIT) {
198 printf("e.type == SDL_QUIT\n");
199 loop_exit();
200
201 } else if (e.type == SDL_KEYDOWN || e.type == SDL_KEYUP) {
202 static uint16_t lastKeyboardKey;
203 SDL_Keymod key_mod = SDL_GetModState() & 4095;
204 tempBtnValues = 0;
205 // printf("key pressed %i - %c\n", e.key.keysym.sym ,e.key.keysym.sym );

207 tempBtnValues = BTN_UP;

211 tempBtnValues = BTN_RIGHT;

221 printf("GOTO prgManagerUI CTRL+ESC\n");
222 loop_exit();
223 prg_change("prgManagerUI");
224 }
225
226 if (e.type == SDL_KEYDOWN && lastKeyboardKey != tempBtnValues) {
227 lastKeyboardKey = tempBtnValues;
228 lastBtnsPressed |= tempBtnValues;

230 lastKeyboardKey = 0;
