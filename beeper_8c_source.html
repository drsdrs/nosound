139void* async( void* arguments ) {
140 float mod_state = 0;
141 uint16_t total_time_us = 0;
142 float tempVal;
143
144 struct async_args* args = arguments;
145 // printf( "Start sync thread ch: %i\n" ,args->channel);
146 async_runnig[args->channel] = true;
147 // printf("args %f - %i - %i\n", args->step_inc, args->step_delay_us, args->channel);
148
149 pthread_setcancelstate( PTHREAD_CANCEL_ENABLE, NULL );
150 pthread_setcanceltype( PTHREAD_CANCEL_ASYNCHRONOUS, NULL );
151 pthread_detach( pthread_self() );
152
153 Beeper_note_set( args->channel, args->note_start );
154 Beeper_pwm_set( args->channel, 1 );
155 while ( total_time_us < args->length_us ) {
156
157 switch ( args->type ) {
158 case PITCH_MOD_LINEAR:
159 chFreq[args->channel] =
160 notePhase[args->note_start] * ( 1 - mod_state ) + notePhase[args->note_end] * mod_state;
161 if ( mod_state >= 1 )
162 total_time_us = args->length_us; // break out of main while loop
163 break;
164 case PITCH_MOD_SINE:
165 tempVal = waves_sine12_get( (uint16_t)( mod_state * 0xfff ) );
166 chFreq[args->channel] = notePhase[args->note_start] * ( 1 - tempVal ) +
167 notePhase[args->note_end] * tempVal;
168 break;
169 case PITCH_MOD_RANDOM:
170 if ( mod_state > 1 ) mod_state--;
171 tempVal = (float)( ( rand() & 0xfff ) * mod_state ) / 0xfff;
172 chFreq[args->channel] = notePhase[args->note_start] * ( 1 - tempVal ) +
173 notePhase[args->note_end] * tempVal;
174 break;
175 }
176
177 SDL_Delay( args->step_delay_us );
178 total_time_us += args->step_delay_us;
179 mod_state += args->step_inc;
180 }
181 chFreq[args->channel] = 0;
182 Beeper_pwm_set( args->channel, 0 );
183 // printf( "EXIT sync thread ch: %i\n" ,args->channel);
184 async_runnig[args->channel] = false;
185 pthread_exit( NULL );
186}
